<script>

	let liElem1 = document.querySelector('#owl');
	let liElem2 = document.querySelector('#salmon');
    let liElem3 = document.querySelector('#tarantula');
    
    function liClickCallback(myEvent) {
    	// first determine which li was clicked.
    	let clickedElement = myEvent.target;
        
        // then get the div-ref id of the linked div
        let divRefId = clickedElement.getAttribute('data-div-ref');
        
        // we take the id and find the element by that id.
        let divRefElement = document.getElementById(divRefId);
        
        // we get the animal-type name of the clickedElement
        let animalType = clickedElement.getAttribute('data-animal-type');
        
        // we put the animal type into the linked div element.
        divRefElement.innerText = animalType;
        console.log(animalType)
    }
    
    liElem1.addEventListener('click', liClickCallback);
    liElem2.addEventListener('click', liClickCallback);
    liElem3.addEventListener('click', liClickCallback);    
        
    
  
</script>
